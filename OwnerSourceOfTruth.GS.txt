/***** Issue #53 – מקור אמת לבעל המערכת *****/

/**
 * מחזיר את מייל בעל המערכת ממנגנון ההגדרות הקיים.
 * סדר העדיפות נשמר מצומצם וברור:
 * 1. הגדרה פעילה מסוג "מערכת" ובמפתח "מייל בעל מערכת".
 * 2. הגדרה פעילה מסוג "מיתוג" באותו מפתח, לתאימות למבנה ההגדרות הקיים.
 * 3. הערך הקשיח הישן, כתאימות לאחור בלבד.
 */
function BH_OWNER_email_() {
  const configured = נרמול_מייל_Build7_(
    getSettingValue_("מערכת", "מייל בעל מערכת") ||
    getSettingValue_("מיתוג", "מייל בעל מערכת") ||
    ""
  );

  return configured || נרמול_מייל_Build7_(BUILD7_OWNER_EMAIL);
}

/**
 * מחזיר את שם בעל המערכת בלי לקבע אותו במסלול הכניסה.
 * אם קיים איש קשר עם מייל הבעלים, שמו הוא מקור האמת.
 * לאחר מכן נבדקת הגדרה מפורשת, ולבסוף נשמר fallback תואם למצב הקיים.
 */
function BH_OWNER_name_() {
  const ownerEmail = BH_OWNER_email_();

  try {
    const contact = readSheet_(SHEETS.CONTACTS).find(c =>
      נרמול_מייל_Build7_(c["אימייל"]) === ownerEmail
    );
    if (contact && String(contact["שם מלא"] || "").trim()) {
      return String(contact["שם מלא"]).trim();
    }
  } catch (e) {
    Logger.log(e);
  }

  return String(
    getSettingValue_("מערכת", "שם בעל מערכת") ||
    getSettingValue_("מיתוג", "שם בעל מערכת") ||
    "אברהם"
  ).trim() || "אברהם";
}

/**
 * בונה את אובייקט בעל המערכת הקנוני עבור מנגנון הכניסה.
 * מחזיר null כאשר המייל המבוקש אינו מייל הבעלים.
 */
function BH_OWNER_authorizedUser_(email) {
  const normalizedEmail = נרמול_מייל_Build7_(email);
  if (!normalizedEmail || normalizedEmail !== BH_OWNER_email_()) return null;

  return {
    allowed: true,
    email: normalizedEmail,
    name: BH_OWNER_name_(),
    role: "מנהל ראשי",
    contactId: "",
    isOwner: true
  };
}

/**
 * פונקציית התקנה ידנית להגדרת מקור האמת בגיליון ההגדרות הקיים.
 * אינה נדרשת בכל כניסה ואינה משנה נתונים אוטומטית.
 */
function BH_OWNER_setOwner_(email, name) {
  email = נרמול_מייל_Build7_(email);
  name = String(name || "").trim();
  if (!BH_INV_isValidEmail_(email)) throw new Error("מייל בעל המערכת אינו תקין");

  BH_OWNER_upsertSetting_("מערכת", "מייל בעל מערכת", email);
  if (name) BH_OWNER_upsertSetting_("מערכת", "שם בעל מערכת", name);

  return { email: email, name: name || BH_OWNER_name_() };
}

function BH_OWNER_upsertSetting_(type, key, value) {
  const rows = readSheet_(SHEETS.SETTINGS);
  const prefix = key + "=";
  const existing = rows.find(r =>
    String(r["סוג הגדרה"] || "").trim() === type &&
    String(r["ערך"] || "").indexOf(prefix) === 0
  );

  const row = existing || {
    "סוג הגדרה": type,
    "ערך": "",
    "פעיל": "כן"
  };

  row["ערך"] = prefix + value;
  row["פעיל"] = "כן";
  upsertSetting_(row);
  return true;
}
