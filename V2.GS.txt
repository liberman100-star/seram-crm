function שמירת_הערת_משימה(data) {
  const actor = דרוש_כתיבה_Build11_2_(data && data.authToken);
  validateRequired_(data.taskId, "משימה");
  validateRequired_(data.note, "הערה");
  הבטח_עמודה_אם_חסרה_(SHEETS.TASK_NOTES, "יוצר");

  const taskForNote = readSheet_(SHEETS.TASKS).find(t => String(t["מזהה משימה"] || "") === String(data.taskId || ""));
  if (taskForNote && taskForNote["מזהה פרויקט"]) BH_AD_assertProjectVisible_(actor, taskForNote["מזהה פרויקט"]);
  const row = {
    "מזהה הערה": makeId_("הע"),
    "מזהה משימה": data.taskId,
    "תאריך ושעת עדכון": new Date(),
    "הערה": data.note,
    "יוצר": BH_FINAL_userName_(actor)
  };

  appendObject_(SHEETS.TASK_NOTES, row);

  const task = readSheet_(SHEETS.TASKS).find(t => t["מזהה משימה"] === data.taskId);
  if (task) {
    task["עדכון אחרון"] = new Date();
    upsert_(SHEETS.TASKS, "מזהה משימה", data.taskId, task);
  }

  log_("שמירת הערת משימה", data.taskId);
  return true;
}

/* הגדרות */

function שמירת_הגדרה(data) {
  validateRequired_(data.type, "סוג הגדרה");
  validateRequired_(data.value, "ערך");

  const rows = readSheet_(SHEETS.SETTINGS);
  const existing = rows.find(r => r["סוג הגדרה"] === data.type && r["ערך"] === data.value);

  if (existing) {
    existing["פעיל"] = data.active || "כן";
    upsertSetting_(existing);
  } else {
    appendObject_(SHEETS.SETTINGS, {
      "סוג הגדרה": data.type,
      "ערך": data.value,
      "פעיל": data.active || "כן"
    });
  }

  log_("שמירת הגדרה", data.type + " | " + data.value);
  return true;
}

function השבתת_הגדרה(type, value) {
  const rows = readSheet_(SHEETS.SETTINGS);
  const row = rows.find(r => r["סוג הגדרה"] === type && r["ערך"] === value);
  if (!row) return false;

  row["פעיל"] = row["פעיל"] === "כן" ? "לא" : "כן";
  upsertSetting_(row);
  return true;
}

function בניית_כותרת_אירוע_יומן_(task) {
  const type = task["סוג משימה"] || "משימה";
  const title = task["כותרת"] || "";
  let target = "";

  if (task["מזהה פרויקט"]) {
    const project = readSheet_(SHEETS.PROJECTS).find(p => p["מזהה פרויקט"] === task["מזהה פרויקט"]);
    if (project) target = "פרויקט " + (project["שם פרויקט"] || "");
  }

  if (!target && task["מזהה איש קשר"]) {
    const contact = readSheet_(SHEETS.CONTACTS).find(c => c["מזהה איש קשר"] === task["מזהה איש קשר"]);
    if (contact) target = "איש קשר " + (contact["שם מלא"] || "");
  }

  return target ? `${type} - ${target} - ${title}` : `${type} - ${title}`;
}

function מחיקת_אירוע_יומן_אם_קיים_(eventId) {
  try {
    const ev = CalendarApp.getCalendarById(CONFIG.CALENDAR_ID).getEventById(eventId);
    if (ev) ev.deleteEvent();
  } catch (e) {}
}

function buildDateTime_(dateValue, timeValue) {
  const d = new Date(dateValue);
  if (isNaN(d.getTime())) throw new Error("תאריך לא תקין");

  const parts = String(timeValue || "09:00").split(":");
  d.setHours(Number(parts[0] || 9), Number(parts[1] || 0), 0, 0);
  return d;
}

function הבטח_עמודה_אם_חסרה_(sheetName, headerName) {
  const sh = getSS_().getSheetByName(sheetName);
  if (!sh) return false;
  const lastCol = Math.max(sh.getLastColumn(), 1);
  const headers = sh.getRange(1, 1, 1, lastCol).getValues()[0];
  if (!headers.includes(headerName)) {
    sh.getRange(1, lastCol + 1).setValue(headerName);
    try { styleHeader_(sh); } catch (e) {}
  }
  return true;
}

function activeSettings_(rows) {
  const out = {};
  rows.forEach(r => {
    if (r["פעיל"] !== "כן") return;
    const type = r["סוג הגדרה"];
    if (!out[type]) out[type] = [];
    out[type].push(r["ערך"]);
  });
  return out;
}

function getSettingValue_(type, key) {
  const row = readSheet_(SHEETS.SETTINGS).find(r =>
    r["סוג הגדרה"] === type &&
    r["ערך"].startsWith(key + "=") &&
    r["פעיל"] === "כן"
  );

  if (!row) return "";
  return String(row["ערך"]).split("=").slice(1).join("=");
}

function calcDashboard_(projects, contacts, tasks) {
  projects = (projects || []).filter(p => String(p["בארכיון"] || "") !== "כן");
  contacts = (contacts || []).filter(c => String(c["בארכיון"] || "") !== "כן");
  tasks = (tasks || []).filter(t => String(t["בארכיון"] || "") !== "כן");

  const today = new Date();
  today.setHours(0, 0, 0, 0);

  const week = new Date(today);
  week.setDate(week.getDate() + 7);

  const open = tasks.filter(t => t["סטטוס"] !== "בוצע" && t["סטטוס"] !== "בוטל");

  const toDate = t => {
    if (!t["תאריך"]) return null;
    const d = new Date(t["תאריך"]);
    d.setHours(0, 0, 0, 0);
    return d;
  };

  return {
    projectsCount: projects.length,
    contactsCount: contacts.length,
    tasksCount: tasks.length,
    lateTasks: open.filter(t => { const d = toDate(t); return d && d < today; }),
    todayTasks: open.filter(t => { const d = toDate(t); return d && d.getTime() === today.getTime(); }),
    weekTasks: open.filter(t => { const d = toDate(t); return d && d > today && d <= week; }),
    activeProjects: projects.slice(0, 5)
  };
}

function שדרוג_מבנה_הגדרות_ערכים_() {
  const ss = getSS_();
  const sh = ss.getSheetByName("הגדרות_ערכים");
  if (!sh) throw new Error("לא נמצא גיליון הגדרות_ערכים");

  const wantedHeaders = [
    "מזהה ערך",
    "סוג ערך",
    "סדר קטגוריה",
    "ערך",
    "פעיל",
    "סדר ערך"
  ];

  const oldData = sh.getDataRange().getValues();
  const oldHeaders = oldData[0] || [];

  const rows = oldData.slice(1).filter(r => r.some(c => c !== ""));

  const categoryOrder = {};
  let nextCategoryOrder = 1;

  const newRows = rows.map(r => {
    const oldObj = {};
    oldHeaders.forEach((h, i) => oldObj[h] = r[i]);

    const type = oldObj["סוג ערך"] || "";
    if (!categoryOrder[type]) {
      categoryOrder[type] = nextCategoryOrder++;
    }

    return [
      oldObj["מזהה ערך"] || makeId_("ערך"),
      type,
      oldObj["סדר קטגוריה"] || categoryOrder[type],
      oldObj["ערך"] || "",
      oldObj["פעיל"] || "כן",
      oldObj["סדר ערך"] || oldObj["סדר"] || 999
    ];
  });

  sh.clear();
  sh.setRightToLeft(true);
  sh.getRange(1, 1, 1, wantedHeaders.length).setValues([wantedHeaders]);

  if (newRows.length) {
    sh.getRange(2, 1, newRows.length, wantedHeaders.length).setValues(newRows);
  }

  styleHeader_(sh);
  sh.autoResizeColumns(1, wantedHeaders.length);
}

}function קבלת_קטגוריות_לבחירה_Build4_4() {
  const rows = readSheet_("הגדרות_ערכים");

  const map = {};
  rows.forEach(r => {
    const type = r["סוג ערך"];
    if (!type) return;

    if (!map[type]) {
      map[type] = {
        name: type,
        categoryOrder: Number(r["סדר קטגוריה"] || 999),
        values: []
      };
    }

    if (r["פעיל"] === "כן") {
      map[type].values.push({
        id: r["מזהה ערך"],
        value: r["ערך"],
        order: Number(r["סדר ערך"] || 999),
        active: r["פעיל"]
      });
    }
  });

  return Object.values(map)
    .sort((a, b) => a.categoryOrder - b.categoryOrder)
    .map(cat => ({
      ...cat,
      values: cat.values.sort((a, b) => a.order - b.order)
    }));
}

function ensureColumnTextFormat_(sheetName, header) {
  const sh = getSS_().getSheetByName(sheetName);
  if (!sh) return;
  const headers = sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0];
  const col = headers.indexOf(header) + 1;
  if (col > 0) sh.getRange(1, col, Math.max(sh.getMaxRows(), 1), 1).setNumberFormat("@");
}

function normalizePhone_(v) {
  let s = String(v || "").trim();
  s = s.replace(/[\s\-]/g, "");
  if (/^5\d{8}$/.test(s)) s = "0" + s;
  if (/^\+9725\d{8}$/.test(s)) s = "0" + s.slice(4);
  if (/^9725\d{8}$/.test(s)) s = "0" + s.slice(3);
  return s;
}

function קבלת_קטגוריות_לבחירה_Build4_4() {
  const rows = readSheet_("הגדרות_ערכים");
  const map = {};
  rows.forEach(r => {
    const type = r["סוג ערך"];
    if (!type) return;
    if (!map[type]) map[type] = { name: type, categoryOrder: Number(r["סדר קטגוריה"] || 999), values: [] };
    map[type].categoryOrder = Math.min(map[type].categoryOrder, Number(r["סדר קטגוריה"] || 999));
    if (r["ערך"]) map[type].values.push({ id: r["מזהה ערך"], value: r["ערך"], order: Number(r["סדר ערך"] || 999), active: r["פעיל"] || "כן" });
  });
  return Object.values(map).sort((a,b)=>a.categoryOrder-b.categoryOrder).map(cat => ({...cat, values: cat.values.sort((a,b)=>a.order-b.order)}));
}

function שמירת_איש_קשר_Build6_3(data) {
  validateRequired_(data.name, "שם מלא");

  הוספת_עמודה_אם_חסרה_(SHEETS.CONTACTS, "הרשאת מערכת");
  הוספת_עמודה_אם_חסרה_(SHEETS.CONTACTS, "פעיל במערכת");
  ensureColumnTextFormat_(SHEETS.CONTACTS, "טלפון");

  const old = data.id ? readSheet_(SHEETS.CONTACTS).find(c => c["מזהה איש קשר"] === data.id) : null;
  const id = data.id || makeId_("איש");

  const permissions = ["ללא הרשאה", "מנהל ראשי", "מנהל", "משתמש", "לקוח"];
  const statuses = ["ליד חדש", "משקיע פוטנציאלי", "לקוח פעיל"];

  BH_SELECTION_VALIDATE_("מקור הגעה", data.source, "מקור הגעה", old ? old["מקור הגעה"] : "");
  BH_SELECTION_VALIDATE_("סוג איש קשר", data.contactType, "סוג איש קשר", old ? old["סוג איש קשר"] : "");
  BH_SELECTION_VALIDATE_("סטטוס לקוח", data.status, "סטטוס לקוח", old ? old["סטטוס לקוח"] : "");

  let status = data.status || (old ? old["סטטוס לקוח"] : "") || "ליד חדש";
  let permission = data.systemPermission || data.permission || (old ? old["הרשאת מערכת"] : "") || "ללא הרשאה";

  // הגנה מפני בלבול של HTML/אימות: אם הרשאה נכנסה בטעות לסטטוס — מעבירים אותה למקום הנכון.
  if (permissions.includes(status)) {
    permission = status;
    status = old && statuses.includes(old["סטטוס לקוח"]) ? old["סטטוס לקוח"] : "ליד חדש";
  }

  if (!statuses.includes(status) && !(old && status === old["סטטוס לקוח"])) status = "ליד חדש";
  if (!permissions.includes(permission)) permission = "ללא הרשאה";

  const row = {
    "מזהה איש קשר": id,
    "שם מלא": data.name,
    "טלפון": שמירת_טלפון_כטקסט_(data.phone),
    "אימייל": data.email || "",
    "כתובת": data.address || "",
    "מקור הגעה": data.source || "",
    "סוג איש קשר": data.contactType || "",
    "סטטוס לקוח": status,
    "תגיות": data.tags || "",
    "אחראי": data.owner || "",
    "הרשאת מערכת": permission,
    "פעיל במערכת": permission === "ללא הרשאה" ? "לא" : "כן",
    "תאריך יצירה": data.created || (old ? old["תאריך יצירה"] : new Date()),
    "עדכון אחרון": new Date()
  };

  upsert_ללא_אימות_(SHEETS.CONTACTS, "מזהה איש קשר", id, row);
  סנכרון_איש_קשר_למשתמש_Build6_2(row);
  log_("שמירת איש קשר", row["שם מלא"]);
  return id;
}

/* הוסר: מופע ישן של function שליחת_קוד_כניסה_Build7 — נשמרה הגרסה המאוחרת בקובץ */
function מספר_בטוח_(value) {
  const n = Number(value);
  return isNaN(n) ? 0 : n;
}

function אימות_קוד_כניסה_Build7(email, code) {
  email = נרמול_מייל_Build7_(email);
  code = String(code || "").trim();
  if (!email || !code) return { ok: false, message: "יש להזין מייל וקוד" };

  const user = איתור_מורשה_לפי_מייל_Build7_(email);
  if (!user.allowed) return { ok: false, message: user.reason };

  const rows = readSheet_(BUILD7_LOGIN_CODES_SHEET);
  const matches = rows.filter(r =>
    נרמול_מייל_Build7_(r["מייל"]) === email &&
    String(r["קוד"] || "").trim() === code &&
    String(r["פעיל"] || "") === "כן" &&
    String(r["נוצל"] || "") !== "כן"
  );

  if (!matches.length) return { ok: false, message: "קוד שגוי או לא פעיל. אם ביקשת קוד חדש, יש להשתמש בקוד האחרון בלבד." };

  // בוחרים את הקוד החדש ביותר לפי Timestamp מספרי, ולא לפי תאריך מפורמט מהגיליון.
  matches.sort((a, b) => מספר_בטוח_(b["יצירה Timestamp"]) - מספר_בטוח_(a["יצירה Timestamp"]));
  const latest = matches[0];

  const expiresMs = מספר_בטוח_(latest["תפוגה Timestamp"]);
  if (!expiresMs) {
    latest["נוצל"] = "כן";
    latest["פעיל"] = "לא";
    upsert_(BUILD7_LOGIN_CODES_SHEET, "מזהה קוד", latest["מזהה קוד"], latest);
    return { ok: false, message: "הקוד נוצר בגרסה ישנה של המערכת. יש לשלוח קוד חדש." };
  }

  if (Date.now() > expiresMs) {
    latest["נוצל"] = "כן";
    latest["פעיל"] = "לא";
    upsert_(BUILD7_LOGIN_CODES_SHEET, "מזהה קוד", latest["מזהה קוד"], latest);
    return { ok: false, message: "הקוד פג תוקף. יש לשלוח קוד חדש." };
  }

  latest["נוצל"] = "כן";
  latest["פעיל"] = "לא";
  upsert_(BUILD7_LOGIN_CODES_SHEET, "מזהה קוד", latest["מזהה קוד"], latest);

  const token = "TOK-" + Utilities.getUuid();
  const now = new Date();
  appendObject_(BUILD7_SESSIONS_SHEET, {
    "Token": token,
    "מייל": email,
    "שם": user.name,
    "תפקיד": user.role,
    "מזהה איש קשר": user.contactId || "",
    "תאריך יצירה": now,
    "יצירה Timestamp": now.getTime(),
    "פעיל": "כן",
    "עדכון אחרון": now
  });

  log_("כניסה למערכת", email + " | " + user.role);
  return { ok: true, token, user };
}

function שמירת_טלפון_כטקסט_(phone) {
  let p = String(phone || "").trim();
  if (!p) return "";
  p = p.replace(/[^\d+]/g, "");
  if (/^5\d{8}$/.test(p)) p = "0" + p;
  if (/^\+9725\d{8}$/.test(p)) p = "0" + p.slice(4);
  if (/^9725\d{8}$/.test(p)) p = "0" + p.slice(3);
  return p;
}

function החלפת_שם_קטגוריה_(oldName, newName) {
  const sh = getSS_().getSheetByName("הגדרות_ערכים");
  if (!sh) return true;
  const data = sh.getDataRange().getValues();
  if (data.length < 2) return true;
  const headers = data[0];
  const typeCol = headers.indexOf("סוג ערך");
  if (typeCol === -1) return true;

  for (let i = 1; i < data.length; i++) {
    if (data[i][typeCol] === oldName) {
      sh.getRange(i + 1, typeCol + 1).setValue(newName);
    }
  }
  return true;
}

function איפוס_סדר_פנימי_לקטגוריות_() {
  const sh = getSS_().getSheetByName("הגדרות_ערכים");
  if (!sh) return true;
  const data = sh.getDataRange().getValues();
  if (data.length < 2) return true;
  const headers = data[0];
  const typeCol = headers.indexOf("סוג ערך");
  const itemOrderCol = headers.indexOf("סדר ערך");
  if (typeCol === -1 || itemOrderCol === -1) return true;

  const groups = {};
  for (let i = 1; i < data.length; i++) {
    const type = data[i][typeCol];
    if (!type) continue;
    if (!groups[type]) groups[type] = [];
    groups[type].push({ row: i + 1, order: Number(data[i][itemOrderCol] || 999) });
  }

  Object.keys(groups).forEach(type => {
    groups[type]
      .sort((a, b) => a.order - b.order)
      .forEach((item, index) => sh.getRange(item.row, itemOrderCol + 1).setValue(index + 1));
  });
  return true;
}

function שמירת_אפשרות_בחירה_Build10(data) {
  validateRequired_(data.category, "קטגוריה");
  validateRequired_(data.item || data.value, "אפשרות בחירה");

  const category = data.category;
  const item = data.item || data.value;
  const rows = readSheet_("הגדרות_ערכים");
  const existing = data.id
    ? rows.find(r => r["מזהה ערך"] === data.id)
    : rows.find(r => r["סוג ערך"] === category && r["ערך"] === item);

  const categoryOrder = data.categoryOrder || קבלת_סדר_קטגוריה_קיים_או_חדש_Build10_(category);
  const itemOrder = data.order || data.itemOrder || קבלת_סדר_הבא_לאפשרות_בחירה_Build10_(category);

  const row = {
    "מזהה ערך": existing ? existing["מזהה ערך"] : makeId_("ערך"),
    "סוג ערך": category,
    "סדר קטגוריה": categoryOrder,
    "ערך": item,
    "פעיל": data.active || (existing ? existing["פעיל"] : "כן"),
    "סדר ערך": itemOrder
  };

  upsert_("הגדרות_ערכים", "מזהה ערך", row["מזהה ערך"], row);
  return true;
}

function קבלת_סדר_קטגוריה_קיים_או_חדש_Build10_(category) {
  const rows = readSheet_("הגדרות_ערכים");
  const existing = rows.find(r => r["סוג ערך"] === category && r["סדר קטגוריה"]);
  if (existing) return Number(existing["סדר קטגוריה"] || 999);
  const max = rows.reduce((m, r) => Math.max(m, Number(r["סדר קטגוריה"] || 0)), 0);
  return max + 1;
}

function קבלת_סדר_הבא_לאפשרות_בחירה_Build10_(category) {
  const rows = readSheet_("הגדרות_ערכים").filter(r => r["סוג ערך"] === category);
  const max = rows.reduce((m, r) => Math.max(m, Number(r["סדר ערך"] || 0)), 0);
  return max + 1;
}

function הוספת_אפשרות_בחירה_בטוח_Build11_(category, value, order) {
  const ss = getSS_();
  let sh = ss.getSheetByName("הגדרות_ערכים");
  if (!sh) {
    sh = ss.insertSheet("הגדרות_ערכים");
    sh.setRightToLeft(true);
    sh.getRange(1,1,1,7).setValues([["מזהה ערך","סוג ערך","ערך","פעיל","סדר קטגוריה","סדר ערך","סדר"]]);
    try { styleHeader_(sh); } catch(e) {}
  }
  const data = sh.getDataRange().getValues();
  const headers = data[0];
  const typeCol = headers.indexOf("סוג ערך");
  const valueCol = headers.indexOf("ערך");
  if (typeCol === -1 || valueCol === -1) return false;
  for (let i=1;i<data.length;i++) {
    if (String(data[i][typeCol]) === category && String(data[i][valueCol]) === value) return true;
  }
  const rowObj = {
    "מזהה ערך": makeId_("ערך"),
    "סוג ערך": category,
    "ערך": value,
    "פעיל": "כן",
    "סדר קטגוריה": קבלת_סדר_קטגוריה_קיים_או_חדש_Build11_(category),
    "סדר ערך": order || 999,
    "סדר": order || 999
  };
  const row = headers.map(h => rowObj[h] !== undefined ? rowObj[h] : "");
  sh.appendRow(row);
  return true;
}

function קבלת_סדר_קטגוריה_קיים_או_חדש_Build11_(category) {
  const rows = readSheet_("הגדרות_ערכים");
  const existing = rows.find(r => r["סוג ערך"] === category && r["סדר קטגוריה"]);
  if (existing) return Number(existing["סדר קטגוריה"] || 999);
  return rows.reduce((m,r)=>Math.max(m, Number(r["סדר קטגוריה"] || 0)), 0) + 1;
}

function נרמול_רשימת_מיילים_Build11_(emails) {
  const seen = {};
  String(emails || "")
    .split(/[;,\n]+/)
    .map(e => String(e || "").trim().toLowerCase())
    .filter(e => e && e.includes("@"))
    .forEach(e => seen[e] = true);
  return Object.keys(seen).join(",");
}

function הוספת_מוזמנים_לאירוע_Build11_(ev, guests) {
  if (!guests) return true;
  guests.split(",").forEach(email => {
    email = email.trim();
    if (!email) return;
    try { ev.addGuest(email); } catch(e) {}
  });
  return true;
}

 * שמירה אינה שולחת זימון; זימון/עדכון נשלח רק מכפתור ייעודי בכרטיס המשימה.

/* הוסר: מופע ישן של function שליחת_זימון_יומן_למשימה_Build11 — נשמרה הגרסה המאוחרת בקובץ */
function סנכרון_מוזמנים_לאירוע_Build11_(ev, guests) {
  const wanted = {};
  נרמול_רשימת_מיילים_Build11_(guests).split(",").filter(Boolean).forEach(e => wanted[e] = true);

  try {
    ev.getGuestList().forEach(g => {
      const email = String(g.getEmail() || "").toLowerCase().trim();
      if (email && !wanted[email]) {
        try { ev.removeGuest(email); } catch(e) {}
      }
    });
  } catch(e) {}

  Object.keys(wanted).forEach(email => {
    try { ev.addGuest(email); } catch(e) {}
  });
  return true;
}

function דרגת_הרשאה_Build11_2_(role) {
  const r = String(role || "ללא הרשאה").trim();
  const map = {"ללא הרשאה":0, "לקוח":1, "משתמש":2, "מנהל":3, "מנהל ראשי":4};
  return map[r] === undefined ? 0 : map[r];

function הרשאה_תפקיד_Build11_2_(role, key) {
  const defaults = {
    "מנהל ראשי": {"דשבורד":"כן","פרויקטים":"כן","אנשי קשר":"כן","משימות":"כן","הערות":"כן","הגדרות":"כן","משתמשים":"כן","הרשאות":"כן","מחיקה":"כן"},
    "מנהל": {"דשבורד":"כן","פרויקטים":"כן","אנשי קשר":"כן","משימות":"כן","הערות":"כן","הגדרות":"לא","משתמשים":"לא","הרשאות":"לא","מחיקה":"כן"},
    "משתמש": {"דשבורד":"כן","פרויקטים":"משויך","אנשי קשר":"משויך","משימות":"שלי","הערות":"פנימי","הגדרות":"לא","משתמשים":"לא","הרשאות":"לא","מחיקה":"לא"},
    "לקוח": {"דשבורד":"לא","פרויקטים":"לקוח בלבד","אנשי קשר":"לא","משימות":"לא","הערות":"לקוח בלבד","הגדרות":"לא","משתמשים":"לא","הרשאות":"לא","מחיקה":"לא"}
  };
  try {
    const row = readSheet_("הרשאות_תפקידים").find(r => String(r["תפקיד"] || "") === String(role || ""));
    if (row && row[key] !== undefined && row[key] !== "") return row[key];
  } catch(e) {}
  return defaults[role] && defaults[role][key] !== undefined ? defaults[role][key] : "לא";
}

function משתמש_מפועל_Build11_2_(token) {
  const user = קבלת_משתמש_מסשן_Build7_(token || "");

function דרוש_כתיבה_Build11_2_(token) {
  const user = משתמש_מפועל_Build11_2_(token);
  const perms = פעולות_מותרות_Build11_2_(user);
  if (!perms.canCreate && !perms.canEdit) throw new Error("אין לך הרשאה לבצע שינוי במערכת");

}

function משתמש_רשאי_לפרויקט_Build11_2_(user, projectId) {
  const role = String(user.role || "").trim();

  if (user.isOwner === true || role === "מנהל ראשי") return true;

  const contactId = String(user.contactId || "").trim();
  const userName = String(user.name || "").trim();

  const project = readSheet_(SHEETS.PROJECTS)
    .find(p => String(p["מזהה פרויקט"]) === String(projectId));

  if (!project) return false;

  if (
    String(project["יוצר"] || "").trim() === userName ||
    String(project["אחראי"] || "").trim() === userName
  ) {
    return true;
  }

  if (BH_AD_userHasProjectDomain_(user, project)) return true;

  const allowed = פרויקטים_מותרים_למשתמש_Build6_3(contactId);
  return allowed.includes(projectId);
}

function משתמש_רשאי_לאיש_קשר_Build11_2_(user, contactId) {
  const role = String(user.role || "").trim();
  if (user.isOwner === true || role === "מנהל ראשי" || role === "מנהל") return true;
  if (user.contactId && user.contactId === contactId) return true;
  const allowed = new Set(פרויקטים_מותרים_למשתמש_Build6_3(user.contactId || ""));
  return readSheet_(SHEETS.LINKS).some(l => l["מזהה איש קשר"] === contactId && allowed.has(l["מזהה פרויקט"]));
}

function סינון_נתונים_לפי_משתמש_Build7_(data, user) {
  const role = String((user && user.role) || "").trim();
  const isSuperAdmin = user && (user.isOwner === true || role === "מנהל ראשי");
  const isManager = role === "מנהל";
  const isClient = role === "לקוח";
  const contactId = String((user && user.contactId) || "").trim();
  const userName = String((user && user.name) || "").trim();

  const out = Object.assign({}, data);
  out.auth = data.auth || { allowed: true };
  out.system = data.system || מיתוג_מערכת_Build7_();
  out.currentUser = data.currentUser || user || {};
  out.currentUser.role = role;
  const actions = פעולות_מותרות_Build11_2_(user);
  out.actionPermissions = actions;
  out.canSeeSettings = actions.canSeeSettings;
  out.canSeeDashboard = actions.canSeeDashboard;
  out.canSeeArchive = actions.canSeeArchive;
  out.canSeeTimeline = actions.canSeeTimeline;
  out.canCreate = actions.canCreate;
  out.canEdit = actions.canEdit;
  out.canDelete = actions.canDelete;
  out.readOnly = actions.readOnly;

  if (isSuperAdmin) {
    out.dashboard = calcDashboard_(out.projects || [], out.contacts || [], out.tasks || []);
    return out;
  }

  if (isManager || role === "משתמש") {
    const linkedProjectIds = פרויקטים_מותרים_למשתמש_Build6_3(contactId);

    out.projects = (data.projects || []).filter(p =>
      linkedProjectIds.includes(p["מזהה פרויקט"]) ||
      String(p["יוצר"] || "").trim() === userName ||
      String(p["אחראי"] || "").trim() === userName ||
      BH_AD_userHasProjectDomain_(user, p)
    );

    const allowedProjectIds = out.projects.map(p => p["מזהה פרויקט"]).filter(Boolean);

    out.links = (data.links || []).filter(l =>
      allowedProjectIds.includes(l["מזהה פרויקט"])
    );

    const allowedContactIds = [...new Set(out.links.map(l => l["מזהה איש קשר"]).filter(Boolean))];

    out.contacts = (data.contacts || []).filter(c =>
      allowedContactIds.includes(c["מזהה איש קשר"]) ||
      c["מזהה איש קשר"] === contactId
    );

    out.tasks = (data.tasks || []).filter(t =>
      allowedProjectIds.includes(t["מזהה פרויקט"]) ||
      t["מזהה איש קשר"] === contactId ||
      String(t["אחראי"] || "").trim() === userName ||
      String(t["יוצר"] || "").trim() === userName
    );

    out.notes = data.notes || [];
    out.canSeeSettings = false;
    out.dashboard = calcDashboard_(out.projects || [], out.contacts || [], out.tasks || []);
    return out;
  }

  if (isClient) {
    const clientLinks = (data.links || []).filter(l =>
      l["מזהה איש קשר"] === contactId &&
      String(l["מורשה צפייה בפורטל"] || "").trim() === "כן"
    );

    const allowedProjectIds = clientLinks.map(l => l["מזהה פרויקט"]).filter(Boolean);

    out.projects = (data.projects || []).filter(p =>
      allowedProjectIds.includes(p["מזהה פרויקט"])
    );
    out.links = clientLinks;
    out.contacts = (data.contacts || []).filter(c => c["מזהה איש קשר"] === contactId);
    out.tasks = [];
    out.notes = [];
    out.realSettings = {};
    out.categories = [];
    out.fields = [];
    out.permissions = {};
    out.canSeeSettings = false;
    out.dashboard = calcDashboard_(out.projects || [], out.contacts || [], out.tasks || []);
    return out;
  }

  out.projects = [];
  out.contacts = [];
  out.links = [];
  out.tasks = [];
  out.notes = [];
  out.realSettings = {};
  out.categories = [];
  out.fields = [];
  out.permissions = {};
  out.canSeeSettings = false;
  out.dashboard = calcDashboard_([], [], []);
  return out;
}


/* הוסר: מופע ישן של function שמירת_פרויקט — נשמרה הגרסה המאוחרת בקובץ */
function שמירת_שיוך(data) {
  דרוש_כתיבה_Build11_2_(data && data.authToken);
  validateRequired_(data.projectId, "פרויקט");
  validateRequired_(data.contactId, "איש קשר");
  הבטח_עמודה_אם_חסרה_(SHEETS.LINKS, "יוצר");
  BH_AD_assertProjectVisible_(משתמש_מפועל_Build11_2_(data && data.authToken), data.projectId);
  const existing = readSheet_(SHEETS.LINKS).find(l =>
    (data.id && l["מזהה שיוך"] === data.id) ||
    (l["מזהה פרויקט"] === data.projectId && l["מזהה איש קשר"] === data.contactId)
  );
  BH_SELECTION_VALIDATE_("תפקיד", data && data.role, "תפקיד בפרויקט", existing ? (existing["תפקיד / כובע"] || existing["תפקיד"] || "") : "");
  if (existing && existing["מזהה שיוך"] !== data.id) throw new Error("איש הקשר כבר משויך לפרויקט הזה");
  const row = {
    "מזהה שיוך": data.id || makeId_("שיך"),
    "מזהה פרויקט": data.projectId,
    "מזהה איש קשר": data.contactId,
    "תפקיד / כובע": data.role || "",
    "מורשה צפייה בפורטל": data.portal || "לא",
    "יוצר": existing && existing["יוצר"] ? existing["יוצר"] : BH_FINAL_userName_(משתמש_מפועל_Build11_2_(data && data.authToken)),
    "תאריך יצירה": data.created || (existing ? existing["תאריך יצירה"] : new Date()),
    "עדכון אחרון": new Date()
  };
  upsert_(SHEETS.LINKS, "מזהה שיוך", row["מזהה שיוך"], row);
  log_("שמירת שיוך", row["מזהה פרויקט"] + " ↔ " + row["מזהה איש קשר"]);
  return true;
}


function BH20_העברה_לארכיון_(sheetName, idField, id, user, label){
  const rows = readSheet_(sheetName);
  const row = rows.find(r => String(r[idField] || "") === String(id || ""));
  if (!row) return true;

  הוספת_עמודה_אם_חסרה_(sheetName, "בארכיון");
  הוספת_עמודה_אם_חסרה_(sheetName, "תאריך ארכיון");
  הוספת_עמודה_אם_חסרה_(sheetName, "הועבר לארכיון על ידי");

  const archivedAt = new Date();
  const archivedBy = BH_FINAL_userName_(user);

  row["בארכיון"] = "כן";
  row["תאריך ארכיון"] = archivedAt;
  row["הועבר לארכיון על ידי"] = archivedBy;
  row["עדכון אחרון"] = archivedAt;

  upsert_ללא_אימות_(sheetName, idField, id, row);

  if(idField === "מזהה פרויקט"){
    הוספת_עמודה_אם_חסרה_(SHEETS.TASKS, "בארכיון");
    הוספת_עמודה_אם_חסרה_(SHEETS.TASKS, "תאריך ארכיון");
    הוספת_עמודה_אם_חסרה_(SHEETS.TASKS, "הועבר לארכיון על ידי");

    const projectName = String(row["שם פרויקט"] || "").trim();

    const tasks = readSheet_(SHEETS.TASKS).filter(t => {
      const taskProject = String(t["מזהה פרויקט"] || "").trim();
      return taskProject === String(id || "").trim() || taskProject === projectName;
    });

    tasks.forEach(t => {
      const taskId = String(t["מזהה משימה"] || "").trim();
      if(!taskId) return;

      t["בארכיון"] = "כן";
      t["תאריך ארכיון"] = archivedAt;
      t["הועבר לארכיון על ידי"] = archivedBy;
      t["עדכון אחרון"] = archivedAt;

      upsert_ללא_אימות_(SHEETS.TASKS, "מזהה משימה", taskId, t);
    });

    log_("העברת משימות פרויקט לארכיון", projectName + " / " + tasks.length);
  }

  log_("העברה לארכיון - " + label, id);
  return true;
}

function מחיקת_פרויקט(arg) {
  const x = חילוץ_מזהה_וטוקן_Build11_2_(arg);
  const user = דרוש_מחיקה_Build11_2_(x.token);

  const project = readSheet_(SHEETS.PROJECTS)
    .find(p => String(p["מזהה פרויקט"]) === String(x.id));

  if (!project) return true;

  const role = String(user.role || "");
  const currentUser = BH_FINAL_userName_(user);

  if (!(user.isOwner === true || role === "מנהל ראשי")) {
    if (role !== "מנהל") throw new Error("אין לך הרשאה להעביר פרויקט לארכיון");
    if (String(project["יוצר"] || "") !== currentUser) {
      throw new Error("ניתן להעביר לארכיון רק פרויקט שאתה יצרת");
    }
  }

  return BH20_העברה_לארכיון_(SHEETS.PROJECTS, "מזהה פרויקט", x.id, user, "פרויקט");
}

function מחיקת_איש_קשר(arg) {

  const contact = readSheet_(SHEETS.CONTACTS)

function BH22_הרשאה_אפקטיבית_(user, key) {

      const row = readSheet_("הרשאות_משתמשים").find(r =>
        String(r["מזהה משתמש"] || "").trim() === contactId &&

function חישוב_שעת_סיום_מטקסט_(dateValue, startTime, minutes) {
  const start = buildDateTime_(dateValue, startTime);
  const end = new Date(start.getTime() + Number(minutes || 60) * 60 * 1000);
  return Utilities.formatDate(end, CONFIG.TIMEZONE, "HH:mm");
}

function buildTaskEndDateTime_(task, start) {
  if (task["סוג משימה"] === "פגישה") {
    if (task["שעת סיום"]) {
      const end = buildDateTime_(task["תאריך"], task["שעת סיום"]);
      if (end <= start) throw new Error("שעת סיום חייבת להיות אחרי שעת התחלה");
      return end;
    }
    const minutes = Number(task["משך פגישה"] || 60);
    return new Date(start.getTime() + minutes * 60 * 1000);
  }

  const minutes = ברירת_מחדל_משך_משימה_(task["סוג משימה"]);
  return new Date(start.getTime() + minutes * 60 * 1000);
}

function syncCalendarEvent_(task) {
  const calendarId = task["מזהה יומן"] || CONFIG.CALENDAR_ID || "primary";

  const cal = CalendarApp.getCalendarById(calendarId);
  if (!cal) throw new Error("לא נמצא יומן Google");

  const start = buildDateTime_(task["תאריך"], task["שעה"]);
  const end = buildTaskEndDateTime_(task, start);

  const title = בניית_כותרת_אירוע_יומן_(task);

  if (task["מזהה אירוע ביומן"]) {
    try {
      const ev = cal.getEventById(task["מזהה אירוע ביומן"]);

      if (ev) {
        ev.setTitle(title);
        ev.setTime(start, end);
        ev.setDescription("נוצר / עודכן אוטומטית ממערכת CRM");
        return ev.getId();
      }
    } catch (e) {}
  }

  const ev = cal.createEvent(title, start, end, {
    description: "נוצר אוטומטית ממערכת CRM"
  });

  return ev.getId();
}

var __BH_SHEET_CACHE__ = {};
function clearSheetCache_(sheetName) {
  if (!__BH_SHEET_CACHE__) __BH_SHEET_CACHE__ = {};
  if (sheetName) delete __BH_SHEET_CACHE__[sheetName];
  else __BH_SHEET_CACHE__ = {};
}

function readSheet_(sheetName) {
  const sh = getSS_().getSheetByName(sheetName);
  if (!sh) return [];

  const lastRow = sh.getLastRow();
  const lastCol = sh.getLastColumn();

  if (lastRow < 2 || lastCol < 1) return [];

  const values = sh.getRange(1, 1, lastRow, lastCol).getValues();
  const headers = values[0].map(h => String(h || "").trim());

  return values.slice(1)
    .filter(r => r.some(c => c !== "" && c !== null))
    .map(r => {
      const obj = {};
      headers.forEach((h, i) => {
        if (!h) return;
        obj[h] = formatValue_(r[i], h);
      });
      return obj;
    });
}

function formatValue_(v, header) {
  if (v === null || v === undefined || v === "") return "";

  if (Object.prototype.toString.call(v) === "[object Date]") {
    if (header === "שעה" || header === "שעת סיום") {
      return Utilities.formatDate(v, CONFIG.TIMEZONE, "HH:mm");
    }

    if (header === "תאריך ושעת עדכון" || header === "תאריך יצירה" || header === "עדכון אחרון") {
      return Utilities.formatDate(v, CONFIG.TIMEZONE, "yyyy-MM-dd HH:mm");
    }

    return Utilities.formatDate(v, CONFIG.TIMEZONE, "yyyy-MM-dd");
  }

  return v;
}

function appendObject_(sheetName, obj) {
  const sh = getSS_().getSheetByName(sheetName);
  const headers = sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0];
  sh.appendRow(headers.map(h => obj[h] !== undefined ? obj[h] : ""));
  clearSheetCache_(sheetName);
}

function upsert_(sheetName, idHeader, id, obj) {
  const sh = getSS_().getSheetByName(sheetName);
  if (!sh) throw new Error("לא נמצא גיליון: " + sheetName);

  const lastCol = sh.getLastColumn();
  if (lastCol < 1) throw new Error("אין כותרות בגיליון: " + sheetName);

  const headers = sh.getRange(1, 1, 1, lastCol).getValues()[0]
    .map(h => String(h || "").trim());

  const idCol = headers.indexOf(idHeader);
  if (idCol === -1) {
    throw new Error("לא נמצאה עמודת מזהה '" + idHeader + "' בגיליון: " + sheetName);
  }

  const row = headers.map(h => obj[h] !== undefined ? obj[h] : "");
  const lastRow = sh.getLastRow();

  if (lastRow >= 2) {
    const ids = sh.getRange(2, idCol + 1, lastRow - 1, 1).getValues().flat();

    for (let i = 0; i < ids.length; i++) {
      if (String(ids[i]) === String(id)) {
        const range = sh.getRange(i + 2, 1, 1, row.length);
        range.clearDataValidations();
        range.setValues([row]);
        clearSheetCache_(sheetName);
        return;
      }
    }
  }

  const newRow = sh.getLastRow() + 1;
  const range = sh.getRange(newRow, 1, 1, row.length);
  range.clearDataValidations();
  range.setValues([row]);
  clearSheetCache_(sheetName);
}

function upsert_ללא_אימות_(sheetName, idHeader, id, obj) {
  return upsert_(sheetName, idHeader, id, obj);
}

function deleteById_(sheetName, idHeader, id) {
  const sh = getSS_().getSheetByName(sheetName);
  const values = sh.getDataRange().getValues();
  const headers = values[0];
  const idCol = headers.indexOf(idHeader);

  for (let i = values.length - 1; i >= 1; i--) {
    if (values[i][idCol] === id) sh.deleteRow(i + 1);
  }
  clearSheetCache_(sheetName);
}

function deleteWhere_(sheetName, header, value) {
  const sh = getSS_().getSheetByName(sheetName);
  const values = sh.getDataRange().getValues();
  const headers = values[0];
  const col = headers.indexOf(header);

  for (let i = values.length - 1; i >= 1; i--) {
    if (values[i][col] === value) sh.deleteRow(i + 1);
  }
  clearSheetCache_(sheetName);
}

function upsertSetting_(row) {
  const sh = getSS_().getSheetByName(SHEETS.SETTINGS);
  const values = sh.getDataRange().getValues();

  for (let i = 1; i < values.length; i++) {
    if (values[i][0] === row["סוג הגדרה"] && values[i][1] === row["ערך"]) {
      sh.getRange(i + 1, 3).setValue(row["פעיל"]);
      clearSheetCache_(SHEETS.SETTINGS);
      return;
    }
  }
  clearSheetCache_(SHEETS.SETTINGS);
}

function BH15_שם_יוצר_(t){

}

function BH15_ערך_שווה_(a,b){
  return String(a || "").trim().toLowerCase() === String(b || "").trim().toLowerCase();

function BH15_משימה_מותרת_ביומן_(t, user, maps, perms){
  const role = String(user.role || "");
  if (user.isOwner === true || role === "מנהל ראשי") return true;

  const userName = String(user.name || "").trim();
  const userEmail = String(user.email || "").trim();

  const creator = BH15_שם_יוצר_(t);

  // משימות ישנות בלי יוצר — לא מסתירים כדי לא לשבור יומן קיים
  if (!creator) return true;

  if (BH15_ערך_שווה_(creator, userName) || BH15_ערך_שווה_(creator, userEmail)) return true;

  const projectId = String(t["מזהה פרויקט"] || "").trim();
  const contactId = String(t["מזהה איש קשר"] || "").trim();
  const projectName = maps.projects[projectId] || "";
  const contactName = maps.contacts[contactId] || "";

  return perms.some(p => {
    const pUser = String(p["משתמש"] || "").trim();
    const type = String(p["סוג הרשאה"] || "").trim();
    const val = String(p["ערך"] || "").trim();

    const userMatch =
      BH15_ערך_שווה_(pUser, userName) ||
      BH15_ערך_שווה_(pUser, userEmail);

    if (!userMatch) return false;

    if (type === "יוצר") {
      return BH15_ערך_שווה_(val, creator);
    }

    if (type === "פרויקט") {
      return BH15_ערך_שווה_(val, projectId) || BH15_ערך_שווה_(val, projectName);
    }

    if (type === "איש קשר") {
      return BH15_ערך_שווה_(val, contactId) || BH15_ערך_שווה_(val, contactName);
    }

    return false;
  });
}

function BH14_העשרת_נתוני_יומן_(data, user){
  data = data || {};
  user = user || {};

  const role = String(user.role || "").trim();
  const isSuperAdmin = user.isOwner === true || role === "מנהל ראשי";
  const userName = String(user.name || "").trim();
  const userEmail = String(user.email || "").trim().toLowerCase();
  const userContactId = String(user.contactId || "").trim();

  const projectsAll = readSheet_(SHEETS.PROJECTS);
  const linksAll = readSheet_(SHEETS.LINKS);

  const projectsById = {};
  projectsAll.forEach(p => {
    projectsById[String(p["מזהה פרויקט"] || "").trim()] = p;
  });

  const allowedProjectIds = {};

  if(!isSuperAdmin){
    linksAll.forEach(l => {
      const contactId = String(l["מזהה איש קשר"] || "").trim();
      const projectId = String(l["מזהה פרויקט"] || "").trim();
      const portal = String(l["מורשה צפייה בפורטל"] || "").trim();

      if(!projectId) return;

      if(contactId === userContactId){
        if(role === "לקוח"){
          if(portal === "כן") allowedProjectIds[projectId] = true;
        } else {
          allowedProjectIds[projectId] = true;
        }
      }
    });

    if(role === "מנהל"){
      projectsAll.forEach(p => {
        const projectId = String(p["מזהה פרויקט"] || "").trim();
        const creator = String(p["יוצר"] || "").trim();
        const owner = String(p["אחראי"] || "").trim();

        if(projectId && (
          BH15_ערך_שווה_(creator, userName) ||
          BH15_ערך_שווה_(creator, userEmail) ||
          BH15_ערך_שווה_(owner, userName) ||
          BH15_ערך_שווה_(owner, userEmail) ||
          BH_AD_userHasProjectDomain_(user, p)
        )){
          allowedProjectIds[projectId] = true;
        }
      });
    }
  }

  function calendarOwnerForTask_(t){
    const projectId = String(t["מזהה פרויקט"] || "").trim();
    const p = projectsById[projectId] || {};

    return String(
      p["יוצר"] ||
      p["אחראי"] ||
      t["יוצר אירוע"] ||
      t["אחראי"] ||
      "ללא יוצר"
    ).trim();
  }

  function hashColorIndex_(s){
    s = String(s || "");
    let n = 0;
    for(let i = 0; i < s.length; i++){
      n = (n + s.charCodeAt(i) * (i + 1)) % 8;
    }
    return n + 1;
  }

  const enrichedTasks = (data.tasks || [])
    .filter(t => String(t["בארכיון"] || "") !== "כן")
    .filter(t => {
      const projectId = String(t["מזהה פרויקט"] || "").trim();
      if(!projectId) return true;

      const p = projectsById[projectId] || {};
      return String(p["בארכיון"] || "") !== "כן";
    })
    .map(t => {
      const copy = Object.assign({}, t);
      const owner = calendarOwnerForTask_(copy);

      copy["יוצר אירוע"] = owner;
      copy["יומן"] = owner;
      copy["תג יומן"] = owner;
      copy["calendarKey"] = owner;
      copy["calendarColorIndex"] = hashColorIndex_(owner);

      return copy;
    });

  data.tasks = enrichedTasks;

  data.calendarTasks = enrichedTasks.filter(t => {
    if(isSuperAdmin) return true;

    const projectId = String(t["מזהה פרויקט"] || "").trim();
    const contactId = String(t["מזהה איש קשר"] || "").trim();

    if(projectId && allowedProjectIds[projectId]) return true;
    if(!projectId && contactId && contactId === userContactId) return true;

    return false;
  });

  data.calendarCreatorsAllowed = [...new Set(
    data.calendarTasks
      .map(t => String(t["יומן"] || t["יוצר אירוע"] || "").trim())
      .filter(Boolean)
  )].sort((a,b)=>a.localeCompare(b, "he"));

  data.calendarCreatorPermission = {
    mode: isSuperAdmin ? "all" : "allowed",
    creators: data.calendarCreatorsAllowed
  };

  return data;
}

function התקנת_הרשאות_יומן_Build15() {
  const ss = getSS_();
  const SHEET_NAME = "הרשאות יומן";

  let sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) sh = ss.insertSheet(SHEET_NAME);

  const headers = [
    "מזהה הרשאה",
    "משתמש",
    "סוג הרשאה",
    "ערך",
    "פעיל",
    "מעתה ואילך",
    "תאריך יצירה",
    "עדכון אחרון"
  ];

  const existingHeaders = sh.getLastColumn()
    ? sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0]
    : [];

  headers.forEach((h) => {
    if (!existingHeaders.includes(h)) {
      sh.getRange(1, sh.getLastColumn() + 1).setValue(h);
    }
  });

  sh.setFrozenRows(1);

  const finalHeaders = sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0];

  const colType = finalHeaders.indexOf("סוג הרשאה") + 1;
  const colActive = finalHeaders.indexOf("פעיל") + 1;
  const colFromNow = finalHeaders.indexOf("מעתה ואילך") + 1;

  if (colType > 0) {
    const rule = SpreadsheetApp.newDataValidation()
      .requireValueInList(["יוצר", "פרויקט", "איש קשר"], true)
      .build();
    sh.getRange(2, colType, Math.max(sh.getMaxRows() - 1, 1), 1).setDataValidation(rule);
  }

  if (colActive > 0) {
    const rule = SpreadsheetApp.newDataValidation()
      .requireValueInList(["כן", "לא"], true)
      .build();
    sh.getRange(2, colActive, Math.max(sh.getMaxRows() - 1, 1), 1).setDataValidation(rule);
  }

  if (colFromNow > 0) {
    const rule = SpreadsheetApp.newDataValidation()
      .requireValueInList(["כן", "לא"], true)
      .build();
    sh.getRange(2, colFromNow, Math.max(sh.getMaxRows() - 1, 1), 1).setDataValidation(rule);
  }

  SpreadsheetApp.flush();
  return "לשונית הרשאות יומן עודכנה בהצלחה";
}

const BH_AD_CATEGORY = "תחום שיוך";
const BH_AD_DEFAULT = "ללא";

function BH_AD_split_(value){
  if (Array.isArray(value)) return value.map(v => String(v || "").trim()).filter(Boolean);
  return String(value || "").split(/[;,|\n]+/).map(v => String(v || "").trim()).filter(Boolean);
}

function BH_AD_join_(values){
  const seen = {};
  return BH_AD_split_(values).filter(v => v !== BH_AD_DEFAULT).filter(v => { if(seen[v]) return false; seen[v] = true; return true; }).join(", ");
}

function BH_AD_projectDomain_(project){
  return String(project && project[BH_AD_CATEGORY] || BH_AD_DEFAULT).trim() || BH_AD_DEFAULT;
}

function BH_AD_managerDomains_(user){
  if(!user) return [];
  const role = String(user.role || "").trim();
  if(role !== "מנהל") return [];
  let raw = user["תחומי שיוך"] || user.assignmentDomains || "";
  if(!raw && user.contactId){
    try {
      const c = readSheet_(SHEETS.CONTACTS).find(x => String(x["מזהה איש קשר"] || "") === String(user.contactId || ""));
      raw = c ? c["תחומי שיוך"] || "" : "";
    } catch(e) {}
  }
  return BH_AD_split_(raw).filter(v => v !== BH_AD_DEFAULT);
}

function BH_AD_userHasProjectDomain_(user, project){
  if(String(user && user.role || "").trim() !== "מנהל") return false;
  const domain = BH_AD_projectDomain_(project);
  if(!domain || domain === BH_AD_DEFAULT) return false;
  return BH_AD_managerDomains_(user).indexOf(domain) !== -1;
}

      "תיקיית Drive": data.drive,
      "תחום שיוך": data.assignmentDomain || data[BH_AD_CATEGORY],
      "יוצר": data.creator
    },
    "משימה": {
      "כותרת": data.title,
      "סוג משימה": data.taskType,
      "תאריך": data.date,
      "שעה": data.time,
      "שעת סיום": data.endTime || data.end,
      "סטטוס": data.status,
      "מזהה פרויקט": data.projectId,
      "מזהה איש קשר": data.contactId,
      "אחראי": data.owner,
      "עדיפות": data.priority,
      "הערות למשימה": data.taskNotes || data.notes
    }
  };

  const key = names[0];
  const map = maps[key] || maps[entity] || {};

  required.forEach(f => {
    const fieldName = String(f["שם שדה"] || "").trim();

    if (!(fieldName in map)) {
      throw new Error("השדה '" + fieldName + "' מוגדר כחובה בישות " + entity + ", אבל אינו מחובר לטופס השמירה");
    }

    if (String(map[fieldName] || "").trim() === "") {
      throw new Error("חובה למלא את השדה: " + fieldName);
    }
  });
}


function BH_SELECTION_OPTION_CATEGORY_(category) {
  const aliases = {
    "תפקיד": "תפקיד / כובע"
  };
  return aliases[category] || category;
}

function BH_SELECTION_ACTIVE_VALUES_(category) {
  const normalized = BH_SELECTION_OPTION_CATEGORY_(category);
  const names = [String(category || "").trim(), normalized].filter(Boolean);
  return readSheet_(SHEETS.SETTINGS)
    .filter(r =>
      names.indexOf(String(r["סוג הגדרה"] || r["סוג ערך"] || "").trim()) !== -1 &&
      String(r["פעיל"] || "כן").trim() !== "לא"
    )
    .map(r => String(r["ערך"] || "").trim())
    .filter(Boolean);
}

function BH_SELECTION_VALIDATE_(category, value, label, oldValue) {
  const submitted = String(value || "").trim();
  if (!submitted) return;

  const previous = String(oldValue || "").trim();
  if (previous && submitted === previous) return;

  let values = BH_SELECTION_ACTIVE_VALUES_(category);
  if (!values.length) {
    try {
      const normalized = BH_SELECTION_OPTION_CATEGORY_(category);
      const names = [String(category || "").trim(), normalized].filter(Boolean);
      values = readSheet_("הגדרות_ערכים")
        .filter(r => names.indexOf(String(r["סוג ערך"] || "").trim()) !== -1 && String(r["פעיל"] || "כן").trim() !== "לא")
        .map(r => String(r["ערך"] || "").trim())
        .filter(Boolean);
    } catch(e) {}
  }
  if (values.indexOf(submitted) === -1) {
    throw new Error((label || category) + " חייב להיות מתוך אפשרויות הבחירה הפעילות בהגדרות");
  }
}

function שחזור_מארכיון(arg){
  arg = arg || {};
  const user = דרוש_מחיקה_Build11_2_(arg.authToken || arg.token);
  const meta = BH21_פרטי_סוג_ארכיון_(arg.type);
  const id = String(arg.id || "").trim();

  if(!id) throw new Error("חסר מזהה לשחזור");

  const row = readSheet_(meta.sheet).find(r => String(r[meta.idField]) === id);
  if(!row) throw new Error(meta.label + " לא נמצא");

  row["בארכיון"] = "לא";
  row["תאריך ארכיון"] = "";
  row["הועבר לארכיון על ידי"] = "";
  row["עדכון אחרון"] = new Date();

  upsert_ללא_אימות_(meta.sheet, meta.idField, id, row);

  if(arg.type === "projects"){
    const tasks = readSheet_(SHEETS.TASKS).filter(t =>
      String(t["מזהה פרויקט"] || "") === id &&
      String(t["בארכיון"] || "") === "כן"
    );

    tasks.forEach(t => {
      const taskId = String(t["מזהה משימה"] || "").trim();
      if(!taskId) return;

      t["בארכיון"] = "לא";
      t["תאריך ארכיון"] = "";
      t["הועבר לארכיון על ידי"] = "";
      t["עדכון אחרון"] = new Date();

      upsert_ללא_אימות_(SHEETS.TASKS, "מזהה משימה", taskId, t);
    });

    log_("שחזור מארכיון - פרויקט כולל משימות", id);
    return true;
  }

  log_("שחזור מארכיון - " + meta.label, id);
  return true;
}

  arg = arg || {};
  דרוש_מחיקה_Build11_2_(arg.authToken || arg.token);

  const meta = BH21_פרטי_סוג_ארכיון_(arg.type);
  const id = String(arg.id || "").trim();

  if(!id) throw new Error("חסר מזהה למחיקה לצמיתות");

  if(arg.type === "projects"){
    const tasks = readSheet_(SHEETS.TASKS).filter(t =>
      String(t["מזהה פרויקט"] || "") === id
    );

    tasks.forEach(t => {
      const taskId = String(t["מזהה משימה"] || "");
      if(t["מזהה אירוע ביומן"]){
        מחיקת_אירוע_יומן_אם_קיים_(t["מזהה אירוע ביומן"]);
      }
      deleteWhere_(SHEETS.TASK_NOTES, "מזהה משימה", taskId);
      deleteById_(SHEETS.TASKS, "מזהה משימה", taskId);
    });

    deleteWhere_(SHEETS.LINKS, "מזהה פרויקט", id);
    deleteById_(SHEETS.PROJECTS, "מזהה פרויקט", id);

    log_("מחיקה לצמיתות מהארכיון - פרויקט", id);
    return true;
  }

  if(arg.type === "contacts"){
    deleteWhere_(SHEETS.LINKS, "מזהה איש קשר", id);
    deleteById_(SHEETS.CONTACTS, "מזהה איש קשר", id);

    log_("מחיקה לצמיתות מהארכיון - איש קשר", id);
    return true;
  }

  if(arg.type === "tasks"){
    const task = readSheet_(SHEETS.TASKS).find(t =>
      String(t["מזהה משימה"] || "") === id
    );

    if(task && task["מזהה אירוע ביומן"]){
      מחיקת_אירוע_יומן_אם_קיים_(task["מזהה אירוע ביומן"]);
    }

    deleteWhere_(SHEETS.TASK_NOTES, "מזהה משימה", id);
    deleteById_(SHEETS.TASKS, "מזהה משימה", id);

    log_("מחיקה לצמיתות מהארכיון - משימה", id);
    return true;
  }

  throw new Error("סוג ארכיון לא תקין");
}

function BH_הבטח_גיליון_עם_כותרות_(sheetName, headers){
  const ss = getSS_();
  let sh = ss.getSheetByName(sheetName);

  if (!sh) {
    sh = ss.insertSheet(sheetName);
  }

  if (sh.getLastRow() === 0) {
    sh.appendRow(headers);
  }

  return sh;
}

function BH_TL_addEvent_(entityType, entityId, eventType, description, userName){
  entityType = String(entityType || "").trim();
  entityId = String(entityId || "").trim();
  eventType = String(eventType || "").trim();
  description = String(description || "").trim();

  if(!entityType || !entityId || !eventType || !description) return;

  BH_TL_ensureTimelineSheet_();

  appendObject_("ציר זמן", {
    "מזהה אירוע": makeId_("אירוע"),
    "סוג ישות": entityType,
    "מזהה ישות": entityId,
    "תאריך ושעה": new Date(),
    "סוג אירוע": eventType,
    "תיאור": description,
    "משתמש": userName || "מערכת"
  });
}

function BH_TL_getTimeline(data){
  data = data || {};

  const user = משתמש_מפועל_Build11_2_(data.authToken || data.token || "");
  const entityType = String(data.entityType || "").trim();
  const entityId = String(data.entityId || "").trim();

  if(!entityType) throw new Error("חסר סוג ישות");
  if(!entityId) throw new Error("חסר מזהה ישות");

  const role = String(user.role || "").trim();
  if(!(user.isOwner === true || role === "מנהל ראשי" || role === "מנהל")){
    throw new Error("אין הרשאה לצפייה בציר הזמן");
  }

  BH_TL_ensureTimelineSheet_();

  return readSheet_("ציר זמן")
    .filter(r =>
      String(r["סוג ישות"] || "") === entityType &&
      String(r["מזהה ישות"] || "") === entityId
    )
    .sort((a,b) =>
      new Date(b["תאריך ושעה"]).getTime() -
      new Date(a["תאריך ושעה"]).getTime()
    );
}
